    <img src="/mnt/data/image.png" style="width:260px;border-radius:12px" />
    <audio id="callTone" src="/mnt/data/call.mp3" loop></audio>
    <button onclick="endCall()" style="margin-top:20px;padding:12px 20px;font-size:18px;border-radius:10px;background:red;color:white;border:none;">Auflegen</button>
  </div>

  <script>
    let display = document.getElementById('display');
    let callScreen = document.getElementById('callScreen');
    let callTone = document.getElementById('callTone');
    let expression = '';

    function press(val){
      if(display.innerText === '0') display.innerText = '';
      display.innerText += val;
      expression = display.innerText;
    }

    function calculate(){
      if(expression === '67'){
        triggerCall();
        return;
      }
      try{
        let result = eval(expression);
        display.innerText = result;
        expression = ''+result;
      } catch{
        display.innerText = 'Fehler';
        expression = '';
      }
    }

    function triggerCall(){
      callScreen.style.display = 'flex';
      callTone.play();
    }

    function endCall(){
      callTone.pause();
      callScreen.style.display = 'none';
      display.innerText = '0';
      expression = '';
    }
  </script>
